def bayesian_blocks(t):
    """Bayesian Blocks Implementation

    By Jake Vanderplas.  License: BSD
    Based on algorithm outlined in http://adsabs.harvard.edu/abs/2012arXiv1207.5578S

    Parameters
    ----------
    t : ndarray, length N
        data to be histogrammed

    Returns
    -------
    bins : ndarray
        array containing the (N+1) bin edges

    Notes
    -----
    This is an incomplete implementation: it may fail for some
    datasets.  Alternate fitness functions and prior forms can
    be found in the paper listed above.
    """
    # copy and sort the array
    t = np.sort(t)
    N = t.size

    # create length-(N + 1) array of cell edges
    edges = np.concatenate([t[:1],
                            0.5 * (t[1:] + t[:-1]),
                            t[-1:]])
    block_length = t[-1] - edges

    # arrays needed for the iteration
    nn_vec = np.ones(N)
    best = np.zeros(N, dtype=float)
    last = np.zeros(N, dtype=int)

    #-----------------------------------------------------------------
    # Start with first data cell; add one cell at each iteration
    #-----------------------------------------------------------------
    for K in range(N):
        # Compute the width and count of the final bin for all possible
        # locations of the K^th changepoint
        width = block_length[:K + 1] - block_length[K + 1]
        count_vec = np.cumsum(nn_vec[:K + 1][::-1])[::-1]

        # evaluate fitness function for these possibilities
        fit_vec = count_vec * (np.log(count_vec) - np.log(width))
        fit_vec -= 4  # 4 comes from the prior on the number of changepoints
        fit_vec[1:] += best[:K]

        # find the max of the fitness: this is the K^th changepoint
        i_max = np.argmax(fit_vec)
        last[K] = i_max
        best[K] = fit_vec[i_max]
    
    #-----------------------------------------------------------------
    # Recover changepoints by iteratively peeling off the last block
    #-----------------------------------------------------------------
    change_points =  np.zeros(N, dtype=int)
    i_cp = N
    ind = N
    while True:
        i_cp -= 1
        change_points[i_cp] = ind
        if ind == 0:
            break
        ind = last[ind - 1]
    change_points = change_points[i_cp:]

    return edges[change_points]

import numpy as np

Imagine you want to track the position of a ball which is being dropped off the Eiffel tower. To measure the ball's position we will setup a camera that can see the whole height of the tower and take many pictures. From the pictures we can then estimate the position of the ball at each of the moments that we took a picture. The Eiffel tower is about 324m heigh. Let's say we have space for 20 pictures in our camera. It takes about 8.5s for the ball to reach the ground, so we will take a picture every 8.5/20 = 0.425seconds.

As the ball is dropping pretty quick it will probably be a bit blurry on our pictures. Plus to fit the whole Eiffel tower in the shot we will have to setup our camera quite a distance away. All this means when we measure the position we will invariably make a small mistake each time. Let's assume we can measure the position with an accuracy of about 10m. The uncertainty is pretty big, but that is on purpose. If we measured the position very accurately in the first place the effect of using the Kalman filter would be much less impressive. The point of this is to show that you can estimate the position of the ball accurately despite only measuring it with a large uncertainty at each time step. Below we plot the true position of the ball in blue and one possible observed path in red. Sometimes it looks like the ball went upwards!

But we can do better! By using what we know about the current position of the ball and Newton's laws we can predict where the next measurement should be. If you combine the prediction and the observation you get a better estimate of the true position then using either one by itself. We could just take the average of the two as our combination, however you can do better than that by weighting the two by how certain you are each one is correct. If the measurement has a large uncertainty and the prediction has a small uncertainty we want to give a larger weight to the prediction, and vice versa. For this we need two ingredients:

  • a method to predict the next position, and
  • a method to estimate how certain we are that the prediction is correct

When the ball is let go at position $x$, $t$ seconds later it will be at $x'=x-\frac{1}{2}\cdot9.81\cdot t^2$. Once it is falling it will gather speed, $v'=v+9.81\cdot t$. As a result it will cover a larger distance during each time step: $x'=x-v\cdot t -\frac{1}{2}\cdot9.81\cdot t^2$.

def kalman(x, P, measurement,
           R=np.matrix("10."),
           motion=np.matrix([-0.5*9.81*time_step**2, -9.81*time_step]).T,
           Q=np.matrix('''0. 0.;
                          0. 0.'''),
           F=np.matrix('''1. %f;
                          0. 1.'''%(time_step)),
           H=np.matrix('''1. 0.''')):
    """Parameters:
    x: initial state
    P: initial uncertainty convariance matrix
    measurement: observed position (same shape as H*x)
    R: measurement noise (same shape as H*x)
    motion: external motion added to state vector x
    Q: motion noise (same shape as P)
    F: next state function: x_prime = F*x
    H: measurement function: position = H*x

    Return: the updated and predicted new values for (x, P)
    """
    # Predict x and P
    x = F*x +motion
    P = F*P*F.T + Q
    print("xp:",x,"Pp:", P)
    
    # Update x, P using measurement m    
    # distance between measured and current position
    y = np.matrix(measurement).T - H * x
    S = H * P * H.T + R  # residual convariance
    K = P * H.T * S.I # Kalman gain or ratio
    print("P:",P)
    print("S.I:",S.I)
    print("H.T:", H.T)
    print("P*H.T:", P*H.T)
    print("H.T*S.I:", H.T*S.I)
    print("P*H.T*S.I:", P*H.T*S.I)
    print("X:",x)
    print("y:",y)
    x = x + K*y
    print("K:",K)
    print("K*y:",K*y)
    I = np.matrix(np.eye(F.shape[0])) # identity matrix
    P = (I - K*H)*P

    return x, P

def next_position(x, v=0.):
    """Given we are at `x`, falling at speed `v` predict the next position and speed"""
    return x - v*time_step - 0.5 * 9.81 * time_step**2, v+9.81*time_step
def position_uncertainty(p, Q=0.5):
    """Estimate uncertainty of our prediction from the uncertainty on current position"""
    # current uncertainty plus a certain amount
    # which takes into account that our prediction
    # is not perfect, this depends strongly on
    # how good you think your extrapolation is
    return p + Q

best_estimate_v = [0.]
estimate_cov_p = [20]
estimate_cov_v = [20]

for n,(obs,t) in enumerate(zip(observations[1:], true_positions[1:])):
    next_p, next_v = next_position(best_estimate_p[n], best_estimate_v[n])
    p_cov = position_uncertainty(estimate_cov_p[n], Q=0)
    
    # this influences if we give more weight to our prediction
    # or more weight to the measurement
    ratio = p_cov/(p_cov + measurement_uncert**2)
    
    # best estimate of the position and uncertainty
    best_p = next_p + ratio*(obs - next_p)
    cov_p = (1-ratio) * p_cov
    best_v = next_v
    best_estimate_p.append(best_p)
    estimate_cov_p.append(cov_p)
    best_estimate_v.append(best_v)
